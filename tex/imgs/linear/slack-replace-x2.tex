\begin{tikzpicture}[
    scale=.7,
    point/.style={circle, fill, inner sep=1.5pt, black},
    axis/.style={->},
    tick/.style={thin},
    grid/.style={
        thin, dotted, teal
    },
    ticklabel/.style={fill=white},
    label/.style={
        fill=white,
        anchor=south west,
        yshift=4pt, xshift=1pt
    },
    legendnode/.style={
        shape=rectangle,
        minimum width= .5cm,
        minimum height= .5cm
    }
]

    \draw[grid]
        (-.5, 0.5) -- (0.5, -.5)
        (-.5, 2.5) -- (2.5, -.5)
        (-.5, 4.5) -- (4.5, -.5)
        (  1, 5  ) -- (  5,   1)
        (  3, 5  ) -- (  5,   3);

    \draw (0,4) -- (3,1);

    \foreach \x in {0, 1, 2, 3, 4}{
        % x2 grid
        \draw [grid] (\x, -.5) -- (\x, 5);
        \coordinate (x_tick\x) at (\x, 0);
        \coordinate (x_tick_a\x) at
            ($(x_tick\x) - (0, 2pt)$);
        \coordinate (x_tick_b\x) at
            ($(x_tick\x) + (0, 2pt)$);
        \draw [tick] (x_tick_a\x) -- (x_tick_b\x);
        \node at (x_tick\x)
            [ticklabel, yshift=-8pt] {$\x$};
    }

    \foreach \s in {4, 2, 0, -2, -4}{
        % center of the tick
        \coordinate (s_tick\s) at
            (-\s/2 + 2, -\s/2 + 2);
        % start of the tick
        \coordinate (s_tick_a\s) at
            ($(s_tick\s) - (+1.5pt, -1.5pt)$);
        % end of the tick
        \coordinate (s_tick_b\s) at
            ($(s_tick\s) - (-1.5pt, +1.5pt)$);
        \draw [tick] (s_tick_a\s) -- (s_tick_b\s);
        \node at (s_tick\s)
            [ticklabel, xshift=-6pt, yshift=6pt, rotate=45] {$\s$};
    }

    \draw[axis]
        (-.5,-.5) -- (5,5)
        node[anchor=west, rotate=45] {$s$};

    \draw[axis]
        (-.5,0) -- (5,0)
        node[anchor=south]{$x_1$};
